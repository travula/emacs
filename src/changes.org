** Insert headers
   User defined tags.  Typing =<head= and =tab= will insert
   the header.  Typing =<prop= and =tab= will insert custom
   id. 
#+BEGIN_EXAMPLE
(require 'org)
(add-to-list 'org-structure-template-alist 
             '("head" 
               "#+TITLE: ?
#+AUTHOR:VLEAD
#+DATE:
#+SETUPFILE: ./org-templates/level-0.org
#+TAGS: boilerplate(b)
#+EXCLUDE_TAGS: boilerplate
#+OPTIONS: ^:nil\n"))

(add-to-list 'org-structure-template-alist
             '("prop"
               ":PROPERTIES:
:CUSTOM_ID: 
:END:"))

#+END_EXAMPLE
